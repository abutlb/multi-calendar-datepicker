شفافية الخلفية: 0.22
💡 الخلفية مرسومة كـ SVG فوانيس في CSS. يمكن استخدام أي صورة: url('images/fanoos.png') أو تدرجات لونية linear-gradient(...). المتغير --mcd-bg-image-opacity يتحكم في درجة الشفافية (0 = مخفي، 1 = كامل).

Code

/* Method 1: Via CSS on the picker element */
.my-picker .mcd-picker {
  --mcd-bg-image:         url('images/fanoos.png');
  --mcd-bg-image-opacity: 0.15; /* 0 = invisible, 1 = fully opaque */
  --mcd-bg-size:          cover;    /* cover | contain | 80px 100px */
  --mcd-bg-repeat:        no-repeat; /* no-repeat | repeat | repeat-x */
  --mcd-bg-position:      center;   /* center | top left | 50% 20% */
}

/* Method 2: Via a theme object */
MultiCalendarDatepicker.init('#ramadan_display', {
  calendar: 'hijri',
  locale:   'ar',
  dir:      'rtl',
  theme: {
    name: 'ramadan',
    vars: {
      '--mcd-bg':                 '#1a0a00',
      '--mcd-text':               '#fde68a',
      '--mcd-primary':            '#f59e0b',
      '--mcd-bg-image':           "url('images/fanoos.png')",
      '--mcd-bg-image-opacity':   '0.15',
      '--mcd-bg-size':            'contain',
      '--mcd-bg-repeat':          'repeat',
      // … other color vars
    },
  },
});

Hijri Options — خيارات الهجري

MultiCalendarDatepicker.init('#birth_display', {
  calendar:       'hijri',
  hijriMode:      'tabular',  // 'tabular' (default) | 'umalqura'
  locale:         'ar',
  dir:            'rtl',
  digits:         'arabic',   // 'arabic' (٠١٢…) | 'latin' (012…)
  outputTarget:   '#birth_value',
  resultTarget:   '#birth_result',
  resultTemplate: 'الميلادي لقاعدة البيانات: {{gregorian}}',
  outputFormat:   'YYYY-MM-DD',
});
