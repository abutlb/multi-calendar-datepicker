Code

MultiCalendarDatepicker.init('#hdate', {
  calendar:     'hijri',   // displayed calendar
  locale:       'ar',
  dir:          'rtl',
  outputTarget: '#hdate_val',  // ← the hidden input
  // No resultTarget = no result div shown
});

// Read the hidden value on change:
document.getElementById('hdate').addEventListener('mcd:change', function(e) {
  console.log(e.detail.gregorian); // '2026-06-15' — always Gregorian ISO
  console.log(e.detail.display);   // '1447/12/19' — Hijri display value
});

Event Detail — تفاصيل الحدث mcd:change

e.detail = {
  gregorian:  '2026-06-15',    // ISO — always Gregorian, always present
  display:    '1447/12/19',    // date in the active calendar system
  year:       1447,            // year in display calendar
  month:      12,
  day:        19,
  monthName:  'ذو الحجة',       // localized month name
  calendar:   'hijri',          // active calendar ID
}
