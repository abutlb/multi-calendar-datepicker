  calendar:       'hijri',
  locale:         'ar',
  dir:            'rtl',
  outputTarget:   '#date_value',   // hidden input → Gregorian ISO
  resultTarget:   '#date_result',  // ← element to fill with result text

Template Variables — متغيرات القالب

{{gregorian}}
Gregorian ISO — لقاعدة البيانات
e.g. 2026-06-15
{{display}}
Date in active calendar
e.g. 1447/12/19
{{year}}
Year (display calendar)
e.g. 1447
{{month}}
Month number
e.g. 12
{{day}}
Day number
e.g. 19
{{monthName}}
Full month name (localized)
e.g. ذو الحجة / December
